package bookstore;

import java.time.LocalDate;

public class BookCompleted implements Comparable<BookCompleted> {
    private final String title;
    private final String author;
    private final LocalDate publishedOn;

    public BookCompleted(String title, String author, LocalDate publishedOn) {
        this.title = title;
        this.author = author;
        this.publishedOn = publishedOn;
    }

    public String getTitle() {
        return title;
    }

    public String getAuthor() {
        return author;
    }

    public LocalDate getPublishedOn() {
        return publishedOn;
    }

    @Override
    public String toString() {
        return "Book{" +
                "title='" + title + '\'' +
                ", author='" + author + '\'' +
                ", publishedOn=" + publishedOn +
                '}';
    }

    @Override
    public int compareTo(BookCompleted that) {
        return this.title.compareTo(that.title);
    }
}

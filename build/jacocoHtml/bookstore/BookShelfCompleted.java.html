package bookstore;

import java.time.Year;
import java.util.*;
import java.util.function.Function;
import java.util.stream.Collectors;

public class BookShelfCompleted {

    private final List<BookCompleted> books = new ArrayList<>();

    public List<BookCompleted> books() {
        return Collections.unmodifiableList(books);
    }

    public void add(BookCompleted... booksToAdd) {
        Arrays.stream(booksToAdd).forEach(books::add);
    }

    public List<BookCompleted> arrange() {
        return arrange(Comparator.naturalOrder());
    }

    public List<BookCompleted> arrange(Comparator<BookCompleted> criteria) {
        return books.stream().sorted(criteria).collect(Collectors.toList());
    }

    public Map<Year, List<BookCompleted>> groupByPublicationYear() {
        return groupBy(book -> Year.of(book.getPublishedOn().getYear()));
    }

    public <K> Map<K, List<BookCompleted>> groupBy(Function<BookCompleted, K> fx) {
        return books
                .stream()
                .collect(Collectors.groupingBy(fx));
    }
}
